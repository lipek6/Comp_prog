#include<bits/stdc++.h>
using namespace std;

// Maybe a greedy algorithm might work? I will kinda use two pointers, but they will be on different vectors
int main (void)
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k; cin >> n >> m >> k;    // "The number of applicants, the number of apartments, and the maximum allowed difference"

    vector<int> a(n);
    vector<int> b(m);
    for(int i = 0; i < n; i++) cin >> a[i];
    for(int i = 0; i < m; i++) cin >> b[i];
    sort(a.begin(), a.end());
    sort(b.begin(), b.end());

    // Maybe running a binary search for each guy who wants an apartment is the way to go?
    int i = 0;
    int j = 0;

    while(i < n && j < m)
    {
        if(abs(a[i] - b[j]) <= 5 && a[i] - b[j] < 0)
        {
            i++;
            j++;
        }
        else if(a[i] - b[j] > 0)
        {
            i++;
            j++;
        }
        else
        {
            j++;
        }
    }
    cout << i + 1 << "\n";
}
