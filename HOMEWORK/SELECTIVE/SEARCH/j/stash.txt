#include<bits/stdc++.h>
using namespace std;

// Maybe an approach from the higher values to the lowers can work, like, get the higher and see if any of the lowers can go with it
int main (void)
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, x; cin >> n >> x;
    vector<long long> p(n);
    for(int i = 0; i < n; i++) cin >> p[i];
    sort(p.begin(), p.end());

    int wheels = 0;
    int k = 0;
    for(int i = n - 1; i >= 0; i--)
    {
        for(int j = k; j < n; j++)
        {
            if(j >= i) 
            {
                wheels++;
                break;
            }

            if(p[i] + p[j] > x)
            {
                if(j - 1 >= 0) k++;
                
                wheels++;
                break;
            }
        }
    }
    cout << wheels << "\n";
}
