#include<bits/stdc++.h>
using namespace std;

int main (void)
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n; cin >> n;

    vector<pair<int,int>> events;
    for(int i = 0; i < n; i++)
    {
        int in, out;
        cin >> in >> out;
        events.push_back({in, 1});
        events.push_back({out+1, -1}); // We can't start to watch another thing at moment 'out' exactly
    }
    sort(events.begin(), events.end());
    

    vector<int> presum(events.size() + 1);
    presum[0] = 0;

    for(int i = 0; i < events.size(); i++)
    {
        presum[i+1] = presum[i] + events[i].second;
        
        if(presum[i+1] >= 3)
        {
            cout << "NO" << "\n";
            return 0;
        }
    }
    cout << "YES" << "\n";
}
